    const output = document.getElementById('output');

    const lastAction = document.getElementById('last-action');

    const sysMode = document.getElementById('sys-mode');

    function appendLine(text, label = '') {

        const time = new Date().toLocaleTimeString();

        output.textContent += `\n[${time}] ${text}`;

        output.scrollTop = output.scrollHeight;

        if (label) lastAction.textContent = 'Última acción: ' + label;

    }

    const simulateDelay = (ms) => new Promise(res => setTimeout(res, ms));

    // BOTÓN: ESTADO

    document.getElementById('btn-status').addEventListener('click', async () => {

        appendLine('Solicitando estado...', 'Estado');

        await simulateDelay(500);

        appendLine('KERNEL: OK | SECTORES: 16 | CONEXIÓN: ESTABLE');

    });

    // BOTÓN: ANALIZAR (INTEGRADO CON TÓTEMS Y CITAS)

    document.getElementById('btn-analyze').addEventListener('click', async () => {

        const nombre = prompt("INTRODUCE VECTOR DE IDENTIDAD (NOMBRE):");

        if (!nombre) return appendLine('ERROR: No se detectó nombre.');

        appendLine(`Analizando frecuencia de: ${nombre.toUpperCase()}...`, 'Analizando');

        sysMode.textContent = 'PROCESANDO';

        await simulateDelay(1000);

        const dna = (nombre.length % 16) || 1;

        const presion = Math.random().toFixed(3);

        const meyi = MAPA_TACTICO[dna] || MAPA_TACTICO[9];

        const res = meyi.totems.find(t => presion <= t.umbral) || meyi.totems[3];

        appendLine('--- RESULTADO DANTE ---');

        appendLine(`MEYI: ${meyi.nombre} | PSI: ${presion}`);

        appendLine(`AVATAR: ${res.animal.toUpperCase()}`);

        appendLine(`CITA: "${res.cita}" — ${res.author || res.autor}`);

        if(presion > 0.618) appendLine('⚠️ ALERTA: MODO RESIDENT EVIL DETECTADO.');

        appendLine('-----------------------');

        sysMode.textContent = 'ACTIVO';

    });

    // BOTÓN: PENSAR (VIRGILIO)

    document.getElementById('btn-think').addEventListener('click', async () => {

        appendLine('Accediendo al pensamiento profundo...', 'Pensando');

        sysMode.textContent = 'MEDITANDO';

        await simulateDelay(1200);

        const randomKey = Object.keys(MAPA_TACTICO)[Math.floor(Math.random() * Object.keys(MAPA_TACTICO).length)];

        sysMode.textContent = 'ACTIVO';

    });

    document.getElementById('btn-copy').addEventListener('click', () => { navigator.clipboard.writeText(output.textContent); appendLine('Log copiado al portapeles.'); });

    document.getElementById('btn-download').addEventListener('click', () => {

        const blob = new Blob([output.textContent], { type: 'text/plain' });

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a'); a.href = url; a.download = 'dante_log.txt'; a.click();

    });
